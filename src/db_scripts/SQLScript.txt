
create database HMS;
use HMS;
create user billing identified by 'hmsbilling';
grant usage on *.* to billing@localhost identified by 'hmsbilling'; 
grant all privileges on HMS.* to billing@localhost; 

create table customer(cid integer, Name varchar(35), Address varchar(35), phonenum varchar(12), emailid varchar(50), Primary Key(cid));


# create table transactiontypes(transtypeid int, transdesc varchar(10));
# CREATE TABLE transactiontypes(transtypeid int AUTO_INCREMENT PRIMARY KEY,transdesc varchar(10));
CREATE TABLE transactiontypes(transtypeid int PRIMARY KEY,transdesc varchar(10));


create table menu_order(oid int, cid int,orderDate Date, transtypeid int, transtypeinfo varchar(20), FOREIGN KEY (transtypeid) REFERENCES transactiontypes(transtypeid));

create table item(itemId integer, itemName varchar(35),categoryId int NOT NULL references categories(categoryId), price float);

create table billing_info(transaction_id int,itemId int references item(itemId), item_instruction varchar(35) );

create table categories(categoryId int, categoryName varchar(35), imageIcon varchar(50));

create table tax(tax1 float, tax2 float, tax3 float);
